* CHUNKS
  #+NAME: pkgver-git
  #+BEGIN_SRC sh
    pkgver() {
        cd "$srcdir/$_gitname"
        printf "%s.r%s.%s.%s" "$pkgver" "$(git rev-list --count HEAD)" "$pkgrel" "$(git rev-parse --short HEAD)"
    }
  #+END_SRC
  #+NAME: build-git
  #+BEGIN_SRC sh
    build() {
        cd "$srcdir/$_gitname"
        mkdir -p build
        cd build
        cmake -DCMAKE_INSTALL_PREFIX=/usr -DUSE_QT5=ON ..
        make
    }
  #+END_SRC
  #+NAME: package-git
  #+BEGIN_SRC sh
    package() {
        cd "$srcdir/$_gitname"
        cd build
        make DESTDIR="$pkgdir" install
    }
  #+END_SRC
